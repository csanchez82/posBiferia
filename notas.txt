CREATE TABLE sucursales (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50),
    pais VARCHAR(50),
    estado VARCHAR(50),
    ciudad VARCHAR(50),
    direccion VARCHAR(50),
    telefono1 VARCHAR(50),
    telefono2 VARCHAR(50),
    correo1 VARCHAR(50),
    correo2 VARCHAR(50),
    notas VARCHAR(250)
);


create table empresa {
    id auto_increment primary key,
    nombre varchar(50),
    web varchar(50),
    logo varchar(50),
};

create table proveedores{
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre varchar(50),
    referencias varchar(50),
    rfc varchar(13),
    PagoDefecto varchar(50),
    banco varchar(50),
    numeroCuenta varchar(20),
    cabe varchar(20),
    swift varchar(20),
    diasCredito int(6),
    creditoDisponible decimal(10,2),
    diasEntrega int(6),
    pais varchar(20),
    estado varchar(20),
    ciudad varchar(20),
    calle varchar(20),
    numero varchar(20),
    colonia varchar(20),
    cp varchar(20),
    correo1 varchar(50),
    correo2 varchar(50),
    telefono1 varchar(50),
    telefono2 varchar(50)
};
